# Crypto (nettle)

# Parallel port (IEEE1284)
option('with_ieee1284',  type: 'feature', value: 'auto', description: 'Enable drivers that require libieee1284 (ieee1284.h)')

# GPIB (linux-gpib / NI)
option('with_gpib',      type: 'feature', value: 'auto', description: 'Enable drivers that require GPIB (gpib/ib.h)')
